Go Package—http_server_header_get_post_data

一、前言

总结记录HTTP服务端获取客户端提交的数据的方式，包括获取HTTP Header、GET、POST等数据。

HTTP Request消息被封装在http.Request结构体中，主要围绕其进行展开。

二、HTTP Header

在http.Request结构体中，使用Header数据结构来存储HTTP头部，其本质是一个map。

// A Header represents the key-value pairs in an HTTP header.
//
// The keys should be in canonical form, as returned by CanonicalHeaderKey.
type Header map[string][]string

  • Header提供的方法有：

    //主要用于http.Client端。
func (h Header) Add(key, value string) 
func (h Header) Set(key, value string) 
func (h Header) Del(key string) 

//主要用于http.Server端获取HTTP头部值。
func (h Header) Get(key string) string
func (h Header) Values(key string) []string

    • 所有方法传入的key均是大小写不敏感（case insensitive）的。

    • 例子：

      func home(w http.ResponseWriter, r *http.Request) {
    fmt.Println(r.Header.Get("host"))  //无效，获取不到值。Host头部不会存储到其中,使用Request.Host。
    fmt.Println(r.Header.Get("content-type"))
    fmt.Println(r.Header.Get("user-agent"))

    fmt.Fprint(w, "Hello Aqua")
}

func main() {
    http.HandleFunc("/", home)
    err := http.ListenAndServe("", nil)
    if err != nil {
        log.Fatalln(err)
    }
}

事实上，对于一些常见的HTTP Header，我们不用直接对Header进行操作，因为http.Request为我们封装了顶层的方法。

  • func (r *Request) UserAgent() string
  • func (r *Request) Referer() string

读取Cookie

  • func (r *Request) Cookies() []*Cookie //获取所有Cookie键值对
  • func (r *Request) Cookie(name string) //获取特定键值对。

其它

  • http.Request.Method
  • http.Request.Host

三、获取GET请求提交的数据

一个URL典型的格式为：

  • [scheme:][//[userinfo@]host][/]path[?query][#fragment]

    type URL struct {
    Scheme     string
    Opaque     string    // encoded opaque data
    User       *Userinfo // username and password information
    Host       string    // host or host:port
    Path       string    // path (relative paths may omit leading slash)
    RawPath    string    // encoded path hint (see EscapedPath method)
    ForceQuery bool      // append a query ('?') even if RawQuery is empty
    RawQuery   string    // encoded query values, without '?'
    Fragment   string    // fragment for references, without '#'
}

  • 获取数据

    // http://localhost/home?name=Aqua&age=18'
func home(w http.ResponseWriter, r *http.Request) {
    // fmt.Println(r.URL.Scheme)
    // fmt.Println(r.URL.Host)
    // fmt.Println(r.URL.Hostname())
    // fmt.Println(r.URL.Port())

    //fmt.Println(r.URL.Path)      //   /home
    //fmt.Println(r.URL.String())  //   /home?name=Aqua&age=18%27

    // fmt.Println(r.URL.RawPath)       // 空
    // fmt.Println(r.URL.EscapedPath()) // /home

    fmt.Println(r.URL.RawQuery) // name=Aqua&age=18%27
    fmt.Println(r.URL.Query())  // map[age:[18'] name:[Aqua]]

    fmt.Fprint(w, "Hello Aqua")
}

func main() {
    http.HandleFunc("/home", home)
    err := http.ListenAndServe("", nil)
    if err != nil {
        log.Fatalln(err)
    }
}

三、获取POST请求的数据

Request结构体使用Form和PostForm存储POST、PUT、PATCH提交的参数，两者区别在于Form会连同URL中的参数也解析存储。两者使用前都必须要先调用Request.ParseForm()方法解析后，才会有数据。

使用MultipartForm存储文件上传的数据，使用前必须要先调用Request.ParseMultipartForm进行解析。

type Request struct {
    ......

    // Form contains the parsed form data, including both the URL
    // field's query parameters and the PATCH, POST, or PUT form data.
    // This field is only available after ParseForm is called.
    // The HTTP client ignores Form and uses Body instead.
    Form url.Values

    // PostForm contains the parsed form data from PATCH, POST
    // or PUT body parameters.
    //
    // This field is only available after ParseForm is called.
    // The HTTP client ignores PostForm and uses Body instead.
    PostForm url.Values

    // MultipartForm is the parsed multipart form, including file uploads.
    // This field is only available after ParseMultipartForm is called.
    // The HTTP client ignores MultipartForm and uses Body instead.
    MultipartForm *multipart.Form

    ......
}

POST、PATCH、PUT提交的参数经过解析后(Request.ParseForm())均会被存储到url.Values中。

// the keys in a Values map are case-sensitive.
type Values map[string][]string

  • func (v Values) Get(key string) string

    可以用Get方法获取参数值。

实际上，http.Request为我们封装了获取的参数的方法，提供的方法会自动先调用Request.ParseForm()，因此可以不用我们自己再去调用了。

  • func (r *Request) FormValue(key string) string
  • func (r *Request) PostFormValue(key string) string

对于文件上传，则是：

  • func (r *Request) FormFile(key string) (multipart.File, *multipart.FileHeader, error)
文本输入：<input type="text">

  • 前端代码：

    <!--home.tmpl-->
<html>
    <head>
        <title>Home Page</title>
    </head>
    <body>
        <form action="/home?name=mikasa" method="post">
            用户名：<input type="text" name="username" />
            密 码：<input type="password" name="password" />
                  <input type="submit" value="Submit" />
        </form>
    </body>
</html>

  • 服务端代码：

    func home(w http.ResponseWriter, r *http.Request) {

    if r.Method == "POST" {

        r.ParseForm()

        //Form包含URL中的参数，PostForm不包含
        fmt.Println(r.Form)                 //map[name:[mikasa] password:[123456] username:[aqua]]
        fmt.Println(r.Form.Get("username")) //aqua
        fmt.Println(r.Form.Get("password")) //123456
        fmt.Println(r.Form.Get("name"))     //mikasa

        fmt.Println(r.PostForm)                 //map[password:[123456] username:[aqua]]
        fmt.Println(r.PostForm.Get("username")) //aqua
        fmt.Println(r.PostForm.Get("password")) //123456
        fmt.Println(r.PostForm.Get("name"))     //""

        //下面两种方法不用调用ParseForm()
        fmt.Println(r.FormValue("username")) //aqua
        fmt.Println(r.FormValue("password")) //123456
        fmt.Println(r.FormValue("name"))     //mikasa

        fmt.Println(r.PostFormValue("username")) //aqua
        fmt.Println(r.PostFormValue("password")) //123456
        fmt.Println(r.PostFormValue("name"))     //""

    }

    t, err := template.ParseFiles("./home.tmpl")
    if err != nil {
        log.Fatalln(err)
    }

    t.Execute(w, nil)
}

func main() {
    http.HandleFunc("/home", home)
    err := http.ListenAndServe("", nil)
    if err != nil {
        log.Fatalln(err)
    }
}

文件上传：<input type="file" />

  • 前端代码

    <!-- home.tmpl -->
<html>
    <head>
        <title>Home Page</title>
    </head>
    <body>
        <form action="/home" method="post" enctype="multipart/form-data">
            头像：<input type="file" name="avatar">
                  <input type="submit"  value="Submit">
        </form>
    </body>
</html>

  • 服务器代码1：

    func home(w http.ResponseWriter, r *http.Request) {

    if r.Method == "POST" {

        r.ParseMultipartForm(1 << 20)  //传入的数值表示最大允许保存在内存的字节数，其它剩余部分会被保存到临时文件中。

        fh := r.MultipartForm.File["avatar"][0] //得到*multipart.FileHeader
        fmt.Println(fh.Filename)                //文件名。
        fmt.Println(fh.Size)                    //文件大小，byte

        f, err := fh.Open() //FileHeader调用Open()方法，可以得到文件句柄。
        if err != nil {
            log.Fatalln(err)
        }

        wf, _ := os.Create(fh.Filename)

        io.Copy(wf, f)

        f.Close()
        wf.Close()

    }

    t, err := template.ParseFiles("./home.tmpl")
    if err != nil {
        log.Fatalln(err)
    }

    t.Execute(w, nil)
}

  • 服务器代码2：

    func home(w http.ResponseWriter, r *http.Request) {

    if r.Method == "POST" {

        //实际上，与代码1本质是一样的，只不过FormFile()方法内部帮我们调用了FileHeader.Open()
        f, fh, err := r.FormFile("avatar")
        if err != nil {
            log.Fatalln(err)
        }

        fmt.Println(fh.Filename)
        fmt.Println(fh.Size)

        wf, _ := os.Create(fh.Filename)

        io.Copy(wf, f)

        f.Close()
        wf.Close()

    }

    t, err := template.ParseFiles("./home.tmpl")
    if err != nil {
        log.Fatalln(err)
    }

    t.Execute(w, nil)
}
